#version 330 core
layout(location = 0) out vec4 color; //Writing to this variable will write to the texture
//out vec4 FragColor;

in vec2 TexCoords;
in vec3 Pos;
in vec3 Normal;
in vec3 worldCameraPos;

uniform sampler2D texture_diffuse1;

void main()
{
    float depth = distance(Pos, worldCameraPos);
    color = vec4(Normal.x, Normal.y, Normal.z, depth); //Refraction gl_FragCoord.z/gl_FragCoord.w
}
